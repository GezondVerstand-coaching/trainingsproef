    // De trainingsschema's opslaan in JavaScript

    const trainingsschema = {

        "dayA": [

            { oefening: "Bench Press", sets: "4", reps: "5-8", rust: "3-5 min" },

            { oefening: "Overhead Press", sets: "3", reps: "6-8", rust: "2-3 min" },

            { oefening: "Squat", sets: "3", reps: "8-10", rust: "3-4 min" }

        ],

        "dayB": [

            { oefening: "Squat", sets: "4", reps: "5-8", rust: "3-5 min" },

            { oefening: "Romanian Deadlift", sets: "3", reps: "8-10", rust: "2-3 min" }

        ],

        "dayC": [

            { oefening: "Deadlift", sets: "4", reps: "4-6", rust: "3-5 min" },

            { oefening: "Lat Pulldown", sets: "3", reps: "6-8", rust: "2-3 min" }

        ]

    };


    function showTrainingDay(dayId) {

        document.querySelectorAll('.training-day').forEach(day => day.style.display = 'none');

        document.getElementById(dayId).style.display = 'block';


        const tableBody = document.getElementById(`tableBody${dayId.charAt(dayId.length - 1).toUpperCase()}`);

        tableBody.innerHTML = "";


        trainingsschema[dayId].forEach((item, index) => {

            const row = document.createElement('tr');

            row.innerHTML = `

                <td>${item.oefening}</td>

                <td>${item.sets}</td>

                <td>${item.reps}</td>

                <td>${item.rust}</td>

                <td><input type="number" id="${dayId}_${index}_weight" placeholder="kg"></td>

                <td><button onclick="saveProgress('${dayId}', ${index})">Opslaan</button></td>

            `;

            tableBody.appendChild(row);

        });

    }


    function showMenu() {

        document.querySelectorAll('.training-day').forEach(day => day.style.display = 'none');

        document.getElementById('menu').style.display = 'block';

    }


    function saveProgress(dayId, exerciseIndex) {

        const weight = document.getElementById(`${dayId}_${exerciseIndex}_weight`).value;

        if (weight) {

            alert(`Voortgang opgeslagen voor ${trainingsschema[dayId][exerciseIndex].oefening}: ${weight} kg`);

            let progressData = JSON.parse(localStorage.getItem("progress")) || {};

            if (!progressData[dayId]) progressData[dayId] = {};

            progressData[dayId][exerciseIndex] = weight;

            localStorage.setItem("progress", JSON.stringify(progressData));

        } else {

            alert("Vul een gewicht in!");

        }

    }


    showMenu();
